    <script>

        // 初始化 CheerpJ，加载当前目录的文件

        cheerpjInit({appPrefix: "."});


        // 捕获 Java 输出到页面

        cheerpjAddPrintCallback(function(text){

            const pre = document.getElementById("output");

            pre.textContent += text + "\n";

        });


        // 运行 DinoGame 主类

        function runGame() {

            cheerpjRunMain("DinoGame", "DinoGame.jar");

        }

    </script>
